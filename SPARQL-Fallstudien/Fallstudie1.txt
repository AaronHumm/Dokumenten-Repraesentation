PREFIX foaf: <http://www.semanticweb.org/jennifertran/ontologies/2018/0/dokumentenRepraesentation#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
SELECT ?dokuName
	WHERE { ?Person foaf:PersonNimmtTeilAnTermin ?Termin .
?Termin foaf:TerminGehoertZuVerwendung ?Verwendung .
?Verwendung foaf:VerwendungHatDokument ?dokuName .
?Person foaf:persNachname ?persNachname .
?Termin foaf:terZeitpunkt ?terZeitpunkt .
FILTER regex( str(?persNachname), 'Tran' )
FILTER regex( str(?terZeitpunkt), '2017-01-27' )
}
